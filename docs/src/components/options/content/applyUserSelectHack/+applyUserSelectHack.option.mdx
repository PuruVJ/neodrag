import Code from '../../OptionsCode.astro';

import UserSelectExample from './UserSelect.example.svelte';

Applies `user-select: none` on `<body />` element when dragging, to prevent the irritating effect where dragging doesn't happen and the text is selected. Applied when dragging starts and removed when it stops.

<Examples>
  <Example>
    <NoUserSelectExample client:visible slot="demo" />

    <Code slot="codes">
      <div slot="svelte">
        ```svelte
        <div use:draggable={{ gpuAcceleration: true }}>
          GPU acceleration on
        </div>
        ```
      </div>

      <div slot="vue">
        ```vue
        <template>
          <div v-draggable="{{ gpuAcceleration: true }}">
            GPU acceleration on
          </div>
        </template>
        ```
      </div>

      <div slot="solid">
        ```jsx
        <div use:draggable={{ gpuAcceleration: true }}>
          GPU acceleration on
        </div>
        ```
      </div>

      <div slot="react">
        ```ts
        useDraggable(draggableRef, { gpuAcceleration: true });
        ```
      </div>

      <div slot="vanilla">
        ```js
        new Draggable(el, { gpuAcceleration: true });
        ```
      </div>

    </Code>

  </Example>
