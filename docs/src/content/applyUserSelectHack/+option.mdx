export const shortDescription = 'Applies `user-select: none` on `<body />` when dragging';

Applies `user-select: none` on `<body />` element when dragging, to prevent the irritating effect where dragging doesn't happen and the text is selected. Applied when dragging starts and removed when it stops.

<Examples client:load>
  <Example>
    <NoUserSelectExample client:visible slot="demo" />

    <Code slot="codes">
      <div slot="svelte">
        ```svelte
        <div use:draggable={{ applyUserSelectHack: true }}>
          User Select disabled
        </div>
        ```
      </div>

      <div slot="vue">
        ```vue
        <template>
          <div v-draggable="{{ applyUserSelectHack: true }}">
            User Select disabled
          </div>
        </template>
        ```
      </div>

      <div slot="solid">
        ```jsx
        <div use:draggable={{ applyUserSelectHack: true }}>
          User Select disabled
        </div>
        ```
      </div>

      <div slot="react">
        ```ts
        useDraggable(draggableRef, { applyUserSelectHack: true });
        ```
      </div>

      <div slot="vanilla">
        ```js
        new Draggable(el, { applyUserSelectHack: true });
        ```
      </div>

    </Code>

  </Example>
