export const shortDescription = 'Axis on which the element can be dragged on.';

{shortDescription} Valid values: `both`, `x`, `y`, `none`.

- `both` - Element can move in any direction
- `x` - Only horizontal movement possible
- `y` - Only vertical movement possible
- `none` - No movement at all

        ```svelte
          Both directions
        </div>
        ```
      </div>

      <div slot="vue">
        ```vue
        <template>
          <div v-draggable="{{ axis: 'both' }}">

          </div>
        ```

      </div>

      <div slot="solid">
        ```jsx
        <div use:draggable={{ axis: 'both' }}>
          Both directions
        </div>
        ```
      </div>

        ```ts
        useDraggable(draggableRef, { axis: 'both' });
        ```
      </div>

      <div slot="vanilla">
