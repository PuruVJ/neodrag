export const shortDescription = 'Element inside draggable that does not trigger dragging';

CSS Selector of an element or multiple elements inside the parent node(on which `use:draggable` is applied).
Can be an element or elements too. If it is provided, Trying to drag inside the `cancel` element(s) will prevent dragging.

<Examples client:load>
  <Example>
    <SingleCancelSelectorExample client:visible slot="demo" />

    <Code slot="codes">
      <div slot="svelte">
        ```svelte

          This will drag!

          <div class="cancel">This won't drag</div>
        </div>
        ```
      </div>

      <div slot="vue">
        ```vue
        <template>
          <div v-draggable="{{ cancel: '.cancel' }}">
            This will drag!

            <div class="cancel">This won't drag</div>

        </template>
        ```
      </div>

      <div slot="solid">

      <div slot="react">
