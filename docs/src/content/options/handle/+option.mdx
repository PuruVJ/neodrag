export const shortDescription = 'Element on which dragging should happen';

CSS Selector of an element or multiple elements inside the parent node(on which `use:draggable` is applied).
Can be an element or elements too.
If it is provided, Only clicking and dragging on this element will allow the parent to drag, anywhere else on the parent won't work.

<Examples client:load>
  <Example>
    <SingleHandleSelectorExample client:visible slot="demo" />

    <Code slot="codes">
      <div slot="svelte">
        ```svelte
        <div use:draggable={{ handle: '.handle' }}>
          Won't drag ❌

          <div class="handle">Drag me ✅</div>
        </div>
        ```
      </div>

      <div slot="vue">
        ```vue
        <template>
          <div v-draggable="{{ handle: '.handle' }}">
            Won't drag ❌

            <div class="handle">Drag me ✅</div>
          </div>
        </template>
        ```
      </div>

      <div slot="solid">
        ```jsx
        <div use:draggable={{ handle: '.handle' }}>
          Won't drag ❌

          <div class="handle">Drag me ✅</div>
        </div>
        ```
      </div>

      <div slot="react">
        ```tsx
        useDraggable(draggableRef, { handle: '.handle' });

        <div ref={draggableRef}>
          Won't drag ❌

          <div className="handle">Drag me ✅</div>
        </div>
        ```
      </div>

      <div slot="vanilla">
         ```js
        new Draggable(draggableEl, { handle: '.handle' });
        ```

        ```html
        <div class="draggable">
          Won't drag ❌

          <div class="handle">Drag me ✅</div>
        </div>
        ```
      </div>

    </Code>
