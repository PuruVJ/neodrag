import SquareGridExample from './SquareGrid.example.svelte';
import RectangleGridExample from './RectangleGrid.example.svelte';
import InactiveGridExample from './InactiveGrid.example.svelte';

export const shortDescription =
  'Applies a grid on the page to which the element snaps to when dragging, rather than the default continuous grid.';

{shortDescription}

::: warning

If you're programmatically creating the grid, do not set it to \[0, 0\] ever, that will stop drag at all. Set it to `undefined` to make it continuous once again

:::

<Examples client:load>

  <Example>
    <SquareGridExample client:visible slot="demo" />

    <Code slot="codes">
      <div slot="svelte">
        ```svelte
        <div use:draggable={{ grid: [50, 50] }}>
          Snaps to 50x50 grid
        </div>
        ```
      </div>

      <div slot="vue">
        ```vue
        <template>
          <div v-draggable="{{ grid: [50, 50] }}">
            Snaps to 50x50 grid
          </div>
        </template>
        ```
      </div>

      <div slot="solid">
        ```jsx
        <div use:draggable={{ grid: [50, 50] }}>
          Snaps to 50x50 grid
        </div>
        ```
      </div>

      <div slot="react">
        ```ts
        useDraggable(draggableRef, { grid: [50, 50] });
        ```
      </div>

      <div slot="vanilla">
        ```js
        new Draggable(el, { grid: [50, 50] });
        ```
      </div>

    </Code>

  </Example>
