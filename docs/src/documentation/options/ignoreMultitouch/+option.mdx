export const shortDescription =
  'Ignores touch events with more than 1 touch.';

{shortDescription}
This helps when you have multiple elements on a canvas where you want to implement pinch-to-zoom behaviour.

<Examples client:load>
  <Example>
    <IgnoredMultitouchExample client:visible slot="demo" />

    <Code slot="codes">
      <div slot="svelte">
        ```svelte
        <div use:draggable={{ ignoreMultitouch: true }}>
          Multi touch ignored
        </div>
        ```
      </div>

      <div slot="vue">
        ```vue
        <template>
          <div v-draggable="{{ ignoreMultitouch: true }}">
            Multi touch ignored
          </div>
        </template>
        ```
      </div>

      <div slot="solid">
        ```jsx
        <div use:draggable={{ ignoreMultitouch: true }}>
          Multi touch ignored
        </div>
        ```
      </div>

      <div slot="react">
        ```ts
        useDraggable(draggableRef, { ignoreMultitouch: true });
        ```
      </div>

      <div slot="vanilla">
        ```js
        new Draggable(el, { ignoreMultitouch: true });
        ```
      </div>

    </Code>

  </Example>
